	/*----- constants -----*/

	/*----- state variables -----*/

turn: 1/-1

table: 2D Array
    null/0 => no player at cell
    1/-1 => winning player
    1 => player's turn
    -1 => dealer autoplay

winner:
    null => no winner
    1/-1 => winning player

lose:
    -1 => dealer wins
    'S' => surrender
    'T' => tie game

score (chips):
    player
    dealer
